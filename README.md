# Program to demonstrate logical operators

a = True
b = False

print("Value of a:", a)
print("Value of b:", b)

print("a AND b:", a and b)
print("a OR b:", a or b)
print("NOT a:", not a)
print("NOT b:", not b)

OUTPUT 
Value of a: True
Value of b: False
a AND b: False
a OR b: True
NOT a: False
NOT b: True
